" --- General Settings ---
set nocompatible            " Use Vim features, not old Vi
" --- Plugins ---
call plug#begin('~/.vim/plugged')

" 1. A beautiful dark theme (Gruvbox is a classic for C-devs)
Plug 'morhetz/gruvbox'

" 2. A tricked-out status bar
Plug 'vim-airline/vim-airline'
Plug 'vim-airline/vim-airline-themes'

call plug#end()
set number                  " Show line numbers
set relativenumber          " Relative line numbers for jumping
set colorcolumn=80          " Visual guide for line length
syntax on                   " Enable syntax highlighting
set mouse=a                 " Enable mouse support
set clipboard=unnamedplus   " System clipboard support
set backspace=indent,eol,start " Fix backspace behavior
set laststatus=2            " Always show status bar
set showmatch               " Briefly jump to matching bracket
set hidden                  " Switch buffers without saving
set scrolloff=8           " Keeps 8 lines of context above/below the cursor
set cursorline            " Highlights the line your cursor is currently on
set updatetime=300        " Makes Vim feel much snappier, especially for plugins
set signcolumn=yes        " Adds a gutter on the left (good for future error markers)

" --- Indentation (Standard VS Code / CS136 Style) ---
set expandtab               " Use spaces instead of tabs
set tabstop=2               " 1 tab = 2 spaces
set shiftwidth=2            " Indent by 2 spaces
set autoindent              " Copy indent from previous line
set smartindent             " C-style auto-indenting
filetype plugin indent on   " Language-specific rules

" --- Search ---
set hlsearch                " Highlight search results
set incsearch               " Search as you type
set ignorecase              " Case insensitive...
set smartcase               " ...unless you use a capital letter
set wildmenu                " Visual autocomplete for command menu
set path+=** " Lets you search all subdirectories

" --- Waterloo C-Dev Shortcut ---
" Press F5 to Save, Compile with clang, and Run
autocmd FileType c nnoremap <F5> :w<CR>:!clang -Wall -std=c99 % -o %< && ./%<<CR>

" --- Standard Auto-close Pairs ---
inoremap ( ()<Left>
inoremap [ []<Left>
inoremap " ""<Left>

" --- Smart Apostrophe Logic (Fixed for "don't") ---
function! SmartApostrophe()
    let pre_char = getline('.')[col('.') - 2]
    if pre_char =~ '\w'
        return "'"
    endif
    return "''\<Left>"
endfunction
inoremap <expr> ' SmartApostrophe()

" --- Smart Curly Brace Logic (VS Code Style) ---
" Typing { gives you {}. Hitting Enter inside {} expands it.
inoremap { {}<Left>
inoremap <expr> <CR> getline('.')[col('.')-2:col('.')-1]=='{}' ? "\<CR>\<Esc>O" : "\<CR>"

" --- Quick Navigation (Like VS Code) ---
" Map Ctrl+s to save (Normal and Insert mode)
nnoremap <C-s> :w<CR>
inoremap <C-s> <Esc>:w<CR>a
inoremap jk <Esc>
inoremap kj <Esc>
" --- Plugin Configurations ---

" Gruvbox Theme Settings
set background=dark         " Tell Vim we are using a dark background
colorscheme gruvbox         " Enable the theme

" Airline Status Bar Settings
let g:airline_theme='gruvbox'
let g:airline#extensions#tabline#enabled = 1  " Shows open files at the top like browser tabs






This one is for VS code:

" === VS Code Vim Optimal Settings ===

" --- Leader Key ---
" Space is universally considered the most comfortable leader key.
" It sits right under your thumb.
let mapleader = " "

" --- Core Navigation & Search ---
set clipboard=unnamedplus   " Share clipboard natively with your OS
set scrolloff=8             " Keep 8 lines of context above/below cursor
set ignorecase              " Search ignores case...
set smartcase               " ...unless you type a capital letter
set hlsearch                " Highlight search results
set incsearch               " Highlight dynamically as you type

" --- The 'Escape' Bindings ---
" As requested, the smoothest way out of Insert mode.
inoremap jk <Esc>
inoremap kj <Esc>

" --- Quality of Life: Centered Jumps ---
" When paging down/up, keep the cursor in the exact middle of the screen.
nnoremap <C-d> <C-d>zz
nnoremap <C-u> <C-u>zz

" When jumping to the next/previous search result, keep it centered.
" (The 'zv' opens any folds if you happen to land inside one).
nnoremap n nzzzv
nnoremap N Nzzzv

" When joining lines (J), keep the cursor at the front instead of jumping to the end.
nnoremap J mzJ`z

" --- Quality of Life: Selection & Indentation ---
" Normally, indenting text in visual mode kicks you back to normal mode.
" This keeps visual mode active so you can indent multiple times easily.
vnoremap < <gv
vnoremap > >gv

" --- File Saving ---
nnoremap <C-s> :w<CR>
inoremap <C-s> <Esc>:w<CR>a

" --- Leader Commands ---
" Clear search highlighting instantly by pressing Space + h
nnoremap <leader>h :nohlsearch<CR>

" --- VS Code Native Compatibility ---
" Force Vim to use VS Code's native undo tree instead of Vim's internal one.
" This prevents VS Code and the Vim extension from getting confused about file history.
nnoremap u :undo<CR>
nnoremap <C-r> :redo<CR>
